I've been playing with the DXML plugin, and have made a couple updates.

DXML.UnloadTable()
- seemed to be messing with tables when unloaded. This seems to work better


Code:

--===========================
-- DXML.UnloadTable
--===========================

function DXML.UnloadTable(...)
IRLUA_PLUGIN_CheckNumArgs(arg,1);
local sXMLTable = IRLUA_PLUGIN_CheckTypes(arg,1,{"string"});
--<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<

if DXML_FILES[sXMLTable] then
	ProcessActive(sXMLTable);
	XML.SetXML("");
	
	for sIndex, sValue in pairs(DXML_FILES[sXMLTable]) do
		DXML_FILES[sXMLTable].sIndex = nil;
	end
	
	DXML_FILES[sXMLTable] = {};
	collectgarbage();
	collectgarbage();
end

end

DXML.ReadFromString() &
DXML.ReadFromFile()
- now re-reads the XML file into memory, very useful if the XML is updated by an external source.

Code:

--===========================
-- DXML.ReadFromFile
--===========================
function DXML.ReadFromFile(...)
IRLUA_PLUGIN_CheckNumArgs(arg,2);
local sFile = IRLUA_PLUGIN_CheckTypes(arg,1,{"string"});
local sXMLTable = IRLUA_PLUGIN_CheckTypes(arg,2,{"string"});
--<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<
if DXML_FILES[sXMLTable] then
	ProcessActive(sXMLTable);
	DXML_FILES[sXMLTable].Name = sXMLTable;
	DXML_FILES[sXMLTable].Path = sFile;
	DXML_FILES[sXMLTable].XML = TextFile.ReadToString(sFile);
	XML.SetXML(DXML_FILES[sXMLTable].XML);
else

DXML_FILES[sXMLTable] = {};
DXML_FILES[sXMLTable].Name = sXMLTable;
DXML_FILES[sXMLTable].Path = sFile;
DXML_FILES[sXMLTable].XML = TextFile.ReadToString(sFile);
end

end

Code:

--===========================
-- DXML.ReadFromString
--===========================
function DXML.ReadFromString(...)
IRLUA_PLUGIN_CheckNumArgs(arg,3);
local sFile = IRLUA_PLUGIN_CheckTypes(arg,1,{"string"});
local sXMLTable = IRLUA_PLUGIN_CheckTypes(arg,2,{"string"});
local sString = IRLUA_PLUGIN_CheckTypes(arg,3,{"string"});
--<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<
if DXML_FILES[sXMLTable] then
	ProcessActive(sXMLTable);
	DXML_FILES[sXMLTable].Name = sXMLTable;
	DXML_FILES[sXMLTable].Path = sFile;
	DXML_FILES[sXMLTable].XML = sString;
	XML.SetXML(DXML_FILES[sXMLTable].XML);
else
	DXML_FILES[sXMLTable] = {};
	DXML_FILES[sXMLTable].Name = sXMLTable;
	DXML_FILES[sXMLTable].Path = sFile;
	DXML_FILES[sXMLTable].XML = sString;
end

end
